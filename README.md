


# PLUGIN: 

phonegap-plugin-iRate



# DESCRIPTION :

PhoneGap plugin for customer/client app rating. [iOS ONLY]


# USAGE:

var args = {debug : "NO",
                 promptAtLaunch : "NO",
                 usesUntilPrompt : 5,
                 eventsUntilPrompt : 10,
                 daysUntilPrompt : "3.0f",
                 remindPeriod : "2.0f",
                 appStoreID : your_app_id};

cordova.iRatePlugin.launch(args);
